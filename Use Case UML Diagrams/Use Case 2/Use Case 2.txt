title USE CASE 2

participant User
participant UI
participant Journal
participant Nutrition Calculator
participant Nutritional Database
participant User Data Editor
participant Database

activate User
activate UI
activate Database



#Present the form instead of the loop
#Instead of Nutrition Calculator
#Have Journal and Main View


User->UI:Start Application
loop 
User->UI:Enter Date,Ingredients,Quantity
User<<--UI:Validate the data entry and present it back to the user
User<<--UI:Display option to add more data and save meal entry
end

UI->User Data Editor:Send all data in the User Data Editor
activate User Data Editor
User Data Editor->Database: Store the meal entry in the Database

deactivate User Data Editor

Database-->Nutrition Calculator:Return the meal entry data to the calculator 

activate Nutrition Calculator

UI->Nutritional Database:Query the nutrient information from the Nutritional Databse



activate Nutritional Database
Nutrition Calculator<--Nutritional Database:Return the nutrient data
Nutrition Calculator-->Journal:Calculate Nutritional Value

activate Journal

User<--Journal:Display the calories per each meal saved
deactivate Journal
opt All Nutrient Breakdown
User->UI:Select a particular meal of which you want to know the Nutirent Breakdown of

UI->Database:Query all the data of the meal selected from the database
Database-->>Nutrition Calculator:Return the meal entry data for the selected meal
UI->Nutritional Database:Query the nutrient breakdown for the selected meal
Nutrition Calculator<--Nutritional Database:Return all the Nutrient data
UI<<--Nutrition Calculator:Return all the nutrient breakdown for the selected meal
UI-->User:Display the Nutrient breakdown of the selected meal to the User
end
User->UI: Close the application
